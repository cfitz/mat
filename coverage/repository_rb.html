97   # conducts the search. takes application and form names (strings), the query xml (string) and page number and page size (integers)
 
98     def search(application, form, queryXML, page = 1, page_size = 25)
 
99       xml = Nokogiri::XML(queryXML)
 
100     
101       params = {}
 
102     
103       # first extract the fulltext query, one is present
 
104       fulltext = xml.search('//query[not(@name)]')
 
105       
106       if !fulltext.first.nil? and !fulltext.first.content.empty?
 
107         params[:q] = fulltext.first.content
 
108       else
 
109          params[:q] = "*:*"
 
110       end
 
111       
112       # filter to only look for values that have been tagged for this form
 
113       params[:fq] = ["mdform_tag_field:#{form}"]
 
114       
115       xml.search('//query[@name]').each do |q| 
116         if !q.content.empty?  
117           params[:fq] << "#{q['name']}:#{q.content}" 
118         end 
119       end
 
120 
121       # get response from SOLR
 
122       solr_docs = @solr.paginate(page, page_size, "select", :params => params) 
123       # store this in the hash so we can access it later
 
124       solr_docs["page"] = page.to_s
 
125       solr_docs["page_size"] = page_size.to_s
 
126  
127       # store the rest of the queries
 
128       solr_docs["queries"] = []
 
129       xml.search('//query[@name]').each {|q| solr_docs["queries"] << q['path'] }
 
130     
131     
132       response = build_results(solr_docs)
 
133       return response

137       raise   e.backtrace.join("\n")
